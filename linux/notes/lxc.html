Host configuration

Create a Network Bridge

Create /etc/sysconfig/network-scripts/ifcfg-virbr0 

DEVICE=br0
TYPE=Bridge
ONBOOT=yes
HOTPLUG=yes
NM_CONTROLLED=no
STP=on
DELAY=0
BOOTPROTO=none
IPADDR=${copy from eth0}
NETMASK=${copy eth0}
GATEWAY=${copy from eth0}
DNS1=${copy from eth0}

Edit /etc/sysconfig/network-scripts/ifcfg-eth0 

DEVICE=eth0
ONBOOT="yes"
HOTPLUG=yes
BRIDGE=br0

Create the container

lxc-create --name=mycontainer --template=centos -- --release 6

Edit /var/lib/lxc/c6test1/config 

lxc.network.type = veth
lxc.network.flags = up
lxc.network.link = br0
lxc.network.hwaddr = fe:1f:76:7b:ce:82
lxc.network.ipv4 = 10.100.100.51/24
lxc.network.ipv4.gateway = 10.100.100.1

Start the container

lxc-start -n c6test1

Container configuration

Edit /etc/sysconfig/network-scripts/ifcfg-eth0 

DEVICE=eth0
TYPE="Ethernet"
BOOTPROTO="none"
DEFROUTE="yes"
ONBOOT=yes
HOSTNAME=
NM_CONTROLLED=no
MTU=
DHCP_HOSTNAME=c6test1
IPADDR0="10.100.100.51"
PREFIX0="24"
GATEWAY0="10.100.100.1"
DNS1="10.100.102.122"
DOMAIN="ebill.co.il"
